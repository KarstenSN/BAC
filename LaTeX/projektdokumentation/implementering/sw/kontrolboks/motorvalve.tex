\textit{Funktionsbeskrivelser}
	\begin{table}[H]
		\begin{tabularx}{\textwidth}{| l | Z |} \hline

		Parametre 	& Ingen 					\\\hline
		Returværdi 	& void 					\\\hline
		Beskrivelse & Åbner for motorventilen. Funktionen aflæser  først om ventilen er åben eller lukket. Er ventilen lukket vil funktionen herefter åbne ventilen og returnere. Er ventilen i forvejen åbnet returneres der med det samme.  \\\hline

	\caption{Metodebeskrivelse for \texttt{open}}

\begin{table}[H]
	\begin{tabularx}{\textwidth}{| l | Z |} 	\hline
		Funktion 	& \texttt{close} 			\\\hline
		Parametre	& Ingen 					\\\hline
		Returværdi 	& void 					\\\hline
		Beskrivelse & Lukker for motorventilen. Funktionen aflæser  først om ventilen er åben eller lukket. Er ventilen åben vil funktionen herefter lukke ventilen og returnere. Er ventilen i forvejen lukket returneres der med det samme.  \\\hline 
	\end{tabularx}
	\caption{Metodebeskrivelse for \texttt{close}}
\end{table}

		Parametre 	& Ingen 					\\\hline
		Returværdi 	& void 						\\\hline
		Beskrivelse & Bremser motorventilen således den låses i sin position. \\\hline
	\end{tabularx}
	\caption{Metodebeskrivelse for \texttt{brake}}
\end{table}


\begin{table}[H]
	\begin{tabularx}{\textwidth}{| l | Z |} \hline
		Funktion 	& \texttt{getStatus} \\\hline
		Parametre 	& Ingen \\\hline
		Returværdi 	& uint8\_t \\\hline
		Beskrivelse & Returnerer positionen for motorventilen. Er ventilen lukket vil den returnere \texttt{VALVE\_CLOSED}. Er ventilen åben vil funktionen returnere \texttt{VALVE\_OPEN}. Er ingen af disse stadier opfyldt returneres \texttt{VALVE\_ERROR}.  \\\hline
	\end{tabularx}
	\caption{Metodebeskrivelse for \texttt{getStatus}}
\end{table}
